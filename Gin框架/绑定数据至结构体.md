gin中的 bind 可以很方便的将前端传递过来的数据与结构体进行参数绑定，以及参数校验。





## 一、绑定表单数据至结构体

```go
tag:form

type StructA struct {
    FieldA string `form:"field_a"`
}

type StructB struct {
    NestedStruct StructA
    FieldB string `form:"field_b"`
}

type StructC struct {
    NestedStructPointer *StructA
    FieldC string `form:"field_c"`
}

type StructD struct {
    NestedAnonyStruct struct {
        FieldX string `form:"field_x"`
    }
    FieldD string `form:"field_d"`
}
//结构体中含form标签，使用Bind
func main() {
    var b StructD
	c.ShouldBind(&b)
    
	c.JSON(200, gin.H{
		"x": b.NestedAnonyStruct,
		"d": b.FieldD,
	})
}
```

## 二、绑定查询参数至结构体

```go
//实际请求
http://localhost:8080/query?name=zhangsan&age=23&sex=%E7%94%B7
tag:form

type UserInfo struct {
	Name string `form:"name"`
	Age  int    `form:"age"`
	Sex  string `form:"sex"`
}


func query(c *gin.Context) {
	var userInfo UserInfo
	err := c.ShouldBindQuery(&userInfo)
	if err != nil {
		fmt.Println(err)
		c.JSON(400, gin.H{"msg": "你错了"})
		return
	}
	c.JSON(200, userInfo)
}

func main() {
	router := gin.Default()
	router.GET("/query", query)
	router.Run()
}
```



## 三、绑定URI（路由上面的参数）至结构体

```go
//实际路由
/uri/futao/23/nan
tag:uri

type UserInfo struct {
	Name string `uri:"name"`
	Age  int    `uri:"age"`
	Sex  string `uri:"sex"`
}

router.GET("/uri/:name/:age/:sex", func(c *gin.Context) {
    var userInfo UserInfo
    err := c.ShouldBindUri(&userInfo)
    if err != nil {
       fmt.Println(err)
       c.JSON(400, gin.H{"msg": "你错了"})
       return
    }
    c.JSON(200, userInfo)
})
```
